



SELECT DISTINCT(GlobalMenuItems.MenuItemText) as Ürün,
       efr_Branchs.BranchName as Şube,
       0 AS 'Satış Adeti'
FROM   GlobalMenuItems
       INNER JOIN efr_Branchs
            ON  GlobalMenuItems.BranchID = efr_Branchs.BranchID
       INNER JOIN OrderTransactions
            ON  GlobalMenuItems.BranchID = OrderTransactions.BranchID
            AND GlobalMenuItems.MenuItemKey = OrderTransactions.MenuItemKey
WHERE  GlobalMenuItems.MenuItemKey NOT IN (SELECT MenuItemKey
                                           FROM   OrderTransactions AS OT1
                                           WHERE  (OT1.LineDeleted = 0)
                                                  AND OT1.OrderDateTime >= @date1
                                                  AND ot1.OrderDateTime <= @date2
                                                  AND ot1.@BranchID)

and GlobalMenuItems.@BranchID
GROUP BY
       GlobalMenuItems.MenuItemText,
       efr_Branchs.BranchName
