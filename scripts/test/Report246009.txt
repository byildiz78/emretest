SELECT posProducts.ProductName,
       posProductUnits.UnitName,
       posManafactureFormulaDetail.Amount,
       posManafactureFormulaDetail.OutagePercentage,
       posManafactureFormulaDetail.UnitCost,
       posManafactureFormulaDetail.OutageCost AS Expr1,
       posManafactureFormulaDetail.NetCost AS Expr2,
       posManafactureFormulaDetail.UnitPrice,
       posProducts.PurchasePrice1        AS IncomingPrice,
       posProducts.LastPurchasePrice     AS LastIncomingPrice,
       posProducts.AveragePurchasePrice  AS AvaragePurchasePrice,
       posManafactureFormula.FormulaName,
       posManafactureFormula.GroupName,
       posManafactureFormula.OutageCost,
       posManafactureFormula.FormulaCost,
       posProducts.LastSalePrice            AS KartSatış
FROM   posManafactureFormula
       INNER JOIN posManafactureFormulaDetail
            ON  posManafactureFormula.FormulaID = posManafactureFormulaDetail.FormulaID
       LEFT OUTER JOIN posProducts
            ON  posManafactureFormulaDetail.ProductID = posProducts.ProductID
       LEFT OUTER JOIN posProductUnits
            ON  posManafactureFormulaDetail.ProductUnitID = posProductUnits.UnitID
WHERE  (posManafactureFormula.GroupName <> 'YARI MAMÜLLER')
