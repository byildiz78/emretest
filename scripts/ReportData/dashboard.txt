SELECT COUNT( distinct h.BranchID) AS deger,'Sube Sayisi' AS Baslik FROM OrderHeaders AS h WITH (NOLOCK) WHERE h.OrderDateTime>DATEADD(hour,6,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) and h.OrderDateTime<DATEADD(hour,30,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) AND h.@BranchID AND isnull(h.LineDeleted,0)=0

UNION ALL

SELECT COUNT(h.AutoID)AS deger,'Cek Sayisi' AS Baslik FROM OrderHeaders AS h WITH (NOLOCK) WHERE h.OrderDateTime>DATEADD(hour,6,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) and h.OrderDateTime<DATEADD(hour,30,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) AND h.@BranchID AND isnull(h.LineDeleted,0)=0

UNION ALL

SELECT isnull(round(SUM(h.AmountDue),2),0) AS deger,'Toplam Ciro' AS Baslik FROM OrderHeaders AS h WITH (NOLOCK) WHERE h.OrderDateTime>DATEADD(hour,6,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) and h.OrderDateTime<DATEADD(hour,30,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) AND h.@BranchID AND h.OrderStatus<3 AND isnull(h.LineDeleted,0)=0

UNION ALL

SELECT isnull(round((SUM(h.AmountDue)/COUNT(h.AutoID)),2),0) AS deger,'Ortalama Cek Tutari' AS Baslik FROM OrderHeaders AS h WITH (NOLOCK) WHERE h.OrderDateTime>DATEADD(hour,6,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) and h.OrderDateTime<DATEADD(hour,30,( DATEADD(dd, 0, DATEDIFF(dd, 0, GETDATE())))) AND h.@BranchID AND h.OrderStatus<3 AND isnull(h.LineDeleted,0)=0
