SELECT     posDocumentTypes.TypeName, posOrderHeader.DocumentDateTime, posCurrents.CurrentCode, posCurrents.CurrentName, posCurrents.CurrentFullName, 
                      posCurrents.CustomField1, posCurrents.CustomField2, posCurrents.CustomField3, posCurrents.CustomField4, posCurrents.CustomField5, posCurrents.CustomField6, 
                      posCurrents.CustomField7, posCurrents.CustomField8, posCurrents.CustomField9, posCurrents.CustomField10, posOrderHeader.OrderSerialNo, 
                      posOrderHeader.TotalAmount, posOrderHeader.Total, posOrderHeader.DiscountTotal, posOrderHeader.TaxTotal, posOrderHeader.TaxDiscountTotal, 
                      posOrderHeader.CustomField1 AS Expr1, posOrderHeader.CustomField3 AS Expr2, posOrderHeader.CustomField2 AS Expr3, posOrderHeader.CustomField5 AS Expr4, 
                      posOrderHeader.CustomField4 AS Expr5, posOrderHeader.CustomField6 AS Expr6, posOrderHeader.CustomField7 AS Expr7, posOrderHeader.CustomField8 AS Expr8, 
                      posOrderHeader.CustomField9 AS Expr9, posOrderHeader.CustomField10 AS Expr10
FROM         posCurrents INNER JOIN
                      posOrderHeader ON posCurrents.CurrentID = posOrderHeader.CurrentID INNER JOIN
                      posDocumentTypes ON posOrderHeader.OrderTypeID = posDocumentTypes.DocumentTypeID
                      WHERE     posOrderHeader.DocumentDateTime >= @ilkTarih AND posOrderHeader.DocumentDateTime <= @SonTarih AND posOrderHeader.IsConverted=0 AND  posDocumentTypes.DocumentTypeID<>41
