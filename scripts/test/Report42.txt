SELECT efr_Branchs.BranchName AS [Şube Adı],
       ISNULL(
           CONVERT(CHAR(24), MAX(OrderHeaders.OrderDateTime), 120),
           '-'
       ) AS [Son Satış Zamanı],
       ISNULL(MAX(OrderHeaders.RecordCounter), 0) AS [KayıtID]
FROM   efr_Branchs
       LEFT OUTER JOIN OrderHeaders
            ON  efr_Branchs.BranchID = OrderHeaders.BranchID
WHERE  efr_Branchs.@BranchID
GROUP BY
       efr_Branchs.BranchName
ORDER BY
       [Son Satış Zamanı] DESC
