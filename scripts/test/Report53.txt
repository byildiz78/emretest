SELECT efr_Branchs.BranchName AS ŞUBE,
 OrderTransactions.MenuItemText AS ÜRÜN,
 OrderTransactions.MenuItemGroupText AS GRUP,
 m.CustomField12 AS KATEGORİ,
 OrderTransactions.MenuItemUnitPrice AS [BİRİM FİYAT],
 SUM(OrderTransactions.Quantity) AS MİKTAR,
 OrderTransactions.AddUserName as Personel
FROM OrderTransactions
INNER JOIN OrderHeaders AS h ON h.OrderKey = OrderTransactions.OrderKey AND h.BranchID = OrderTransactions.BranchID AND h.LineDeleted=0
 LEFT OUTER JOIN efr_Branchs ON OrderTransactions.BranchID = efr_Branchs.BranchID
 INNER JOIN posProducts as m WITH (NOLOCK) on m.ProductKey=OrderTransactions.MenuItemKey
WHERE OrderTransactions.TransactionDateTime >= @date1
 AND OrderTransactions.TransactionDateTime <= @date2
 AND OrderTransactions.LineDeleted = 0
AND OrderTransactions.@BranchID 
GROUP BY
 OrderTransactions.AddUserName,
 efr_Branchs.BranchName,
 OrderTransactions.MenuItemText,
 OrderTransactions.MenuItemGroupText,
 m.CustomField12,
 OrderTransactions.MenuItemUnitPrice
Order By
OrderTransactions.AddUserName
