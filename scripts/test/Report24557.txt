
SELECT     'Tüketilen Para Puan' AS Tipi, gsOrderHeaders.CreditAmountUsed AS Tutar, CustomerFiles.CustomerName AS [Müşteri Adı], 
                      CustomerFiles.PhoneNumber AS [Müşteri Telefonu]
FROM         gsOrderHeaders INNER JOIN
                      CustomerCredits ON CustomerCredits.CreditID = gsOrderHeaders.CreditID LEFT OUTER JOIN
                      CustomerFiles ON CustomerFiles.CustomerID = CustomerCredits.CustomerID AND CustomerCredits.BranchID = CustomerFiles.BranchID



Where gsOrderHeaders.OrderDateTime <=@sonTarih And  gsOrderHeaders.OrderDateTime >=@ilkTarih and  gsorderheaders.OrderStatus <>'3'
  and gsOrderHeaders.@BranchID
