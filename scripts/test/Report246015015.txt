SELECT DISTINCT
       efr_Branchs.BranchName            AS [Şube],
       ISNULL(CustomerFiles.CustomerName, 'TANIMSIZ') AS [Müşteri Adı],
       ISNULL(CustomerPhones.PhoneNumber, '') AS Telefon,
       ISNULL(CustomerFiles.EmailAddress, '') AS Email,
  
       CustomerFiles.adddatetime         AS [Kayıt Tarihi]
FROM   CustomerFiles
       LEFT OUTER JOIN CustomerPhones
            ON  CustomerPhones.CustomerKey = CustomerFiles.CustomerKey
       INNER JOIN efr_Branchs
            ON  CustomerFiles.BranchID = efr_Branchs.BranchID
WHERE  CustomerFiles.@BranchID AND len(CustomerPhones.PhoneNumber)>8 --and CustomerPhones.PhoneNumber  like '5%'-- and CustomerPhones.PhoneNumber  like '5%'
ORDER BY
       CustomerFiles.adddatetime            DESC
