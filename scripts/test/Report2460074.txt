SELECT 
	br.BranchName AS [Şube],
	CONVERT(VARCHAR,t.OrderDateTime,103) AS [Tarih],
	CONVERT(VARCHAR,t.OrderDateTime,108) AS [Saat],
	t.AccountingCode AS [Stok Kodu],
	t.MenuItemText [Ürün Adı], 
	t.MenuItemGroupText AS [Grubu], 
	t.MenumItemCategoryText AS [Kategori],
	ROUND(SUM(t.Quantity), 3) AS Adet,
	SUM(t.ExtendedPrice) AS Tutar
FROM OrderTransactions  AS t WITH (NOLOCK)
INNER JOIN efr_Branchs AS br WITH (NOLOCK) ON br.BranchID = t.BranchID
INNER JOIN OrderHeaders h WITH (NOLOCK) ON h.OrderKey = t.OrderKey
WHERE 
	t.OrderDateTime BETWEEN @date1 AND @date2
	AND h.LineDeleted = 0 
	AND t.LineDeleted = 0 
	AND t.@BranchID 
GROUP BY 
	br.BranchName,
	t.AccountingCode,
	t.OrderDateTime,
	t.MenuItemText,
	t.MenuItemGroupText, 
	t.MenumItemCategoryText
