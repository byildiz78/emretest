SELECT 
t.AddUserName as Personel,
b.BranchName as [Şube],
convert(varchar,t.OrderDateTime,104) as Tarih,
SUM(CASE WHEN p.CustomField12='TABAKLAR' THEN t.Quantity ELSE 0 END) [Tabak Sayısı],
SUM(CASE WHEN p.CustomField12 LIKE '%SAĞLIKLI%' THEN t.Quantity ELSE 0 END) [Sağlıklı İçecekler],
SUM(CASE WHEN p.CustomField13 = 'BAŞLANGIÇ' THEN t.Quantity ELSE 0 END) [Başlangıç Ürünleri],
SUM(CASE WHEN p.CustomField12 = 'YAN ÜRÜNLER' THEN t.Quantity ELSE 0 END) [Yan Ürünler]
FROM OrderTransactions AS t WITH(NOLOCK)
INNER JOIN posProducts as p WITH (NOLOCK) on p.ProductKey=t.MenuItemKey
INNER JOIN efr_Branchs b on b.BranchID=t.BranchID
WHERE t.OrderDateTime between @date1 and @date2
AND t.@BranchID
AND t.linedeleted=0
AND t.MenuItemText not like '%İSTEMİYORUM%'
GROUP BY b.BranchName,convert(varchar,t.OrderDateTime,104),
 t.AddUserName
order by b.BranchName,convert(varchar,t.OrderDateTime,104)
