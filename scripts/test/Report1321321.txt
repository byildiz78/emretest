SELECT 
       OrderTransactions.MenuItemText AS Ürün,
       OrderTransactions.Quantity AS Miktar,
       efr_Branchs.BranchName as Sube,
       CONVERT(CHAR(24),OrderTransactions.OrderDateTime,120)  AS Tarih,
       OrderTransactions.OrderID AS [Çek No],
       OrderTransactions.MenuItemText AS Ürün,
       OrderTransactions.Quantity AS Miktar,
       OrderTransactions.ExtendedPrice AS Tutar,
			 (CASE OrderHeaders.OrderTypeSourceID 
			 when '103' then 'PDS'
			 when '101' then 'Yemek Sepeti'
			 when '107' then 'Trendyol'
			 when '104' then 'Getir'
			 when '111' then 'Migros Yemek'
			 when '1' then 'Masa Satış'
       else null end) as SatışYeri,
	   WebIntegrationOrderHeaders.OrderNo as [Sipariş Numarası],
           WebIntegrationOrderHeaders.CustomerName as [Müşteri Adı],
           WebIntegrationOrderHeaders.PhoneNumber as [Telefon Numarası]
FROM   OrderTransactions 
INNER JOIN OrderHeaders ON OrderTransactions.OrderKey=OrderHeaders.OrderKey
LEFT JOIN WebIntegrationOrderHeaders ON OrderTransactions.OrderKey=WebIntegrationOrderHeaders.OrderKey
INNER JOIN efr_Branchs ON OrderHeaders.BranchID=efr_Branchs.BranchID
WHERE  (OrderTransactions.LineDeleted = 0)
       AND OrderTransactions.OrderDateTime >=  @date1
       AND OrderTransactions.OrderDateTime <= @date2
       AND OrderTransactions.@BranchID
       AND OrderHeaders.OrderTypeSourceExternalNo is not null
ORDER BY
       OrderTransactions.OrderID DESC
