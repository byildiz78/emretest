SELECT MenuItems.MenuItemID AS [Ürün Kodu],
 MenuItems.MenuItemText AS [Ürün Adı],
 MenuGroups.MenuGroupText AS Grubu,
 MenuCategories.MenuCategoryText AS Kategorisi,
 OrderTransactions.MenuItemUnitPrice AS [Birim Fiyat],
 SUM(OrderTransactions.Quantity) AS Miktar,
 efr_Branchs.BranchName AS [Şube Adı]
FROM OrderTransactions
 INNER JOIN MenuItems
 ON OrderTransactions.MenuItemID = MenuItems.MenuItemID
 AND OrderTransactions.BranchID = MenuItems.BranchID
 INNER JOIN MenuCategories
 ON MenuItems.MenuCategoryID = MenuCategories.MenuCategoryID
 AND MenuItems.BranchID = MenuCategories.BranchID
 INNER JOIN MenuGroups
 ON MenuItems.BranchID = MenuGroups.BranchID
 AND MenuItems.MenuGroupID = MenuGroups.MenuGroupID
 INNER JOIN OrderHeaders
 ON OrderTransactions.BranchID = OrderHeaders.BranchID
 AND OrderTransactions.OrderID = OrderHeaders.OrderID
 LEFT OUTER JOIN efr_Branchs
 ON OrderTransactions.BranchID = efr_Branchs.BranchID
WHERE (OrderHeaders.OrderDateTime >= @ilkTarih)
 AND (OrderHeaders.OrderDateTime <= @sonTarih)
 AND OrderTransactions.@BranchID
 AND (OrderTransactions.TransactionStatus = '1')
GROUP BY
 MenuItems.MenuItemID,
 MenuItems.MenuItemText,
 MenuGroups.MenuGroupText,
 MenuCategories.MenuCategoryText,
 OrderTransactions.MenuItemUnitPrice,
 efr_Branchs.BranchName
