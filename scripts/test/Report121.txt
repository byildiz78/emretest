DECLARE @startDate DATETIME;set @startDate=GETDATE();
IF DATEPART(hour,@startDate)<6
 set @startDate=DATEADD(dd, -1, DATEDIFF(dd, 0, @startDate))
ELSE
set @startDate=DATEADD(dd, 0, DATEDIFF(dd, 0, @startDate))
SELECT   OrderID AS [Çek No], 
CONVERT(CHAR(24), PaymentDateTime , 120) AS [Ödeme Tarihi], AmountPaid AS [Tutar], (
      CASE PaymentMethod
      WHEN '1' THEN 'NAKIT'
         WHEN '2' THEN 'TICKET'
         WHEN '3' THEN 'VISA'
         WHEN '4' THEN 'SODEXO'
         WHEN '5' THEN 'MULTINET'
         WHEN '6' THEN 'SETCARD'
         WHEN '7' THEN 'VISA 5'
         WHEN '8' THEN 'VISA 6' 
         WHEN '9' THEN 'GRUPANYA'
         ELSE 'Kapanmadı'
      END) AS OdemeTipi
FROM         OrderPayments
Where OrderPayments.PaymentDateTime>=@startDate And OrderPayments.@BranchID
