SELECT     SUM(Orderheaders.AmountDue) AS Tutar,efr_Branchs.BranchName AS [SUBE],sum(guestNumber) as KisiSayisi, SUM(Orderheaders.AmountDue)/nullif (sum(guestNumber),0) as KisiOrt
FROM         Orderheaders INNER JOIN
                      efr_Branchs ON orderheaders.BranchID = efr_Branchs.BranchID
WHERE  orderheaders.orderdatetime >= @date1
                  AND orderheaders.orderdatetime <= @date2
AND orderheaders.@BranchID
                  AND orderheaders.linedeleted=0

GROUP BY efr_Branchs.BranchName  ORDER BY Tutar DESC
