select k.MenuModifierText, SUM(k.Toplam) AS Toplam from (

SELECT
g.MenuModifierText,

(Sum(CASE WHEN t.mod1ID IN (1,2,64) then 1 else 0 end) + Sum(CASE WHEN t.mod2ID IN (1,2,64) then 1 else 0 end)) as Toplam
		

FROM ordertransactions as t
INNER JOIN posproducts as p on p.productkey=t.MenuItemkey
INNER JOIN GlobalMenuModifiers as g on g.MenumodifierID=t.mod1ID
WHERE 
t.OrderDateTime>@date1
AND t.OrderDateTime<@date2
AND t.Linedeleted=0
and t.@branchID
AND p.CustomField12='TABAKLAR'
AND g.MenuModifierText IN ('FULL MAKARNA','FULL SALATA')
Group by g.MenuModifierText

UNION ALL

SELECT  
case when t.notes LIKE '%full makarna%' THEN 'FULL MAKARNA' ELSE ''END AS MenuModifierText,
Sum(CASE WHEN t.notes LIKE '%full makarna%' then 1 else 0 end) AS Toplam
FROM OrderTransactions as t
INNER JOIN GlobalMenuModifiers as g on g.MenumodifierID=t.mod1ID
INNER JOIN posproducts as p on p.productkey=t.MenuItemkey
WHERE 
t.OrderDateTime>@date1
AND t.OrderDateTime<@date2
AND t.Linedeleted=0
AND t.Notes LIKE '%full makarna%'
AND p.CustomField12='TABAKLAR'
and t.@branchID
Group by case when t.notes LIKE '%full makarna%' THEN 'FULL MAKARNA' ELSE ''END

UNION ALL 

SELECT  
case when t.notes LIKE '%full salata%' THEN 'FULL SALATA' ELSE ''END AS MenuModifierText,
Sum(CASE WHEN t.notes LIKE '%full salata%' then 1 else 0 end) AS Toplam
FROM OrderTransactions as t
INNER JOIN GlobalMenuModifiers as g on g.MenumodifierID=t.mod1ID
INNER JOIN posproducts as p on p.productkey=t.MenuItemkey
WHERE 
t.OrderDateTime>@date1
AND t.OrderDateTime<@date2
AND t.Linedeleted=0
AND t.Notes LIKE '%full SALata%'
AND p.CustomField12='TABAKLAR'
and t.@branchID
Group by case when t.notes LIKE '%full salata%' THEN 'FULL SALATA' ELSE ''END

) as k
GROUP BY 
k.MenuModifierText
