SELECT posProducts.ProductCode AS [Stok Kodu],
 posproducts.ExternalCode as [Harici Kodu],
 posProducts.ProductName AS [Stok AdÄ±],
 posproducts.TaxPercent1 AS KDV,
 posProductUnits.UnitName AS [Temel Birim],
 posPickList.PickValue AS Kategori,
 posPickList_1.PickValue AS Grup,
  Case When posProducts.IsActive=1 then 'Aktif'
		  Else 'Pasif' end as [Durum]
FROM posWarehouse
 INNER JOIN posProductPrices
 INNER JOIN posProducts
 ON posProductPrices.ProductID = posProducts.ProductID
 INNER JOIN posProductUnits
 ON posProducts.UnitID = posProductUnits.UnitID
 ON posWarehouse.WarehouseID = posProducts.DefaultWarehouseID
 INNER JOIN posProductUnits AS posProductUnits_1
 ON posProducts.UnitID2 = posProductUnits_1.UnitID
 LEFT OUTER JOIN posPickList
 ON posProducts.CategoryID = posPickList.ListID
 LEFT OUTER JOIN posPickList AS posPickList_1
 ON posProducts.GroupID = posPickList_1.ListID
WHERE posProducts.IsSaleProduct = 1 

ORDER BY
 posProducts.ProductName ASC
