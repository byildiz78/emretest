DECLARE @startDate DATETIME;set @startDate=GETDATE();
IF DATEPART(hour,@startDate)<6
 set @startDate=DATEADD(dd, -1, DATEDIFF(dd, 0, @startDate))
ELSE
set @startDate=DATEADD(dd, 0, DATEDIFF(dd, 0, @startDate))
SELECT OrderHeaders.OrderDateTime, DineInTables.DineInTableText, DineInTableGroups.TableGroupText, OrderHeaders.AmountDue, 
 OrderHeaders.OrderType
FROM DineInTableGroups INNER JOIN
 DineInTables ON DineInTableGroups.TableGroupID = DineInTables.TableGroupID AND 
 DineInTableGroups.BranchID = DineInTables.BranchID INNER JOIN
 OrderHeaders ON DineInTables.BranchID = OrderHeaders.BranchID AND DineInTables.DineInTableID = OrderHeaders.DineInTableID
WHERE (OrderHeaders.OrderType = N'1')
and OrderHeaders.OrderDateTime >=@startDate And OrderHeaders.@BranchID
