SELECT eb.BranchName AS [Şube],  webFormFilledDetail.HeaderID AS [AnketID], webFormDetail.FieldTitle [Soru],webFormFilledDetail.FieldValue AS [Cevap],
webFormFilled.AutoID,webFormFilled.FormNotes AS Müşteri, webFormFilled.AddUserName AS Ekleyen,
 webFormFilled.FormDate AS [Anket Tarihi], webFormFilled.AddDateTime AS [Eklenme Tarihi]
  FROM webFormFilledDetail  
 INNER JOIN webFormDetail ON webFormFilledDetail.FormID = webFormDetail.FormID AND webFormFilledDetail.DetailID = webFormDetail.AutoID 
 INNER JOIN webFormFilled ON webFormFilled.AutoID = webFormFilledDetail.HeaderID
 INNER JOIN efr_Branchs eb ON eb.BranchID = webFormFilled.BranchID
 WHERE webFormFilled.FormDate>=@date1 AND webFormFilled.FormDate<@date2 
 AND webFormFilled.@BranchID AND webFormFilledDetail.FormID=1
ORDER BY webFormFilledDetail.HeaderID, webFormFilledDetail.DetailID
