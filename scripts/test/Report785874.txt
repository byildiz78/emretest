SELECT
	BR.BranchName AS [Şube],
	EK.StationEcrNo AS [Yazar Kasa],
	EK.DataUsedPercent AS [Kullanılan  Data Alan %],
	EK.DataFreePercent AS [Serbest Data Alan %],

Case when 
 (EK.DataUsedPercent -2) * 2 >100
then (EK.DataUsedPercent -3) * 2 

else  (EK.DataUsedPercent -2) * 2  end 

 as  [Ekü Doluluk Yüzde],
	

CONVERT(VARCHAR, EK.AddDatetime, 104) + ' ' + CONVERT(VARCHAR, EK.AddDatetime, 24) AS [Tarih]
FROM
	GmpEkuReports EK
LEFT JOIN efr_Branchs BR ON BR.BranchID = EK.BranchID

WHERE EK.AutoID IN ( SELECT MAX(AutoID) FROM GmpEkuReports G WHERE G.DataFreePercent < 100.0 GROUP BY BranchID, StationEcrNo )

ORDER BY EK.DataFreePercent ASC
