select 
--OrderType AS PAKET,
br.BranchName AS ŞUBE,
t.MainMenuItemText AS COMBO,
t.MenuItemText as ÜRÜN,
t.Quantity AS ADET,
--h.orderdatetime TARİH,
convert(varchar, h.OrderDateTime, 103) tarih

 from OrderTransactions as t with(nolock)
INNER JOIN Orderheaders as h with(nolock) on h.OrderKey=t.OrderKey
RIGHT JOIN efr_Branchs as br on br.BranchID=h.BranchID
 where h.OrderDateTime BETWEEN @date1 and @date2 
AND t.MenuItemText like '%FINDIKLI FIRIN SÜTLAÇ%' 
and t.MainMenuItemText like '%KARNE  HEDİYE%'
--and h.OrderType=5 
AND h.@BranchID
and h.LineDeleted=0
and t.LineDeleted=0

ORDER BY h.BranchID asc
