SELECT EmployeeFiles.FirstName+' '+isnull(EmployeeFiles.LastName,'') AS CustomerName,
       OrderPayments.AmountPaid,
       OrderPayments.PaymentDateTime,
       OrderPayments.OrderID
FROM   OrderPayments
       LEFT OUTER JOIN EmployeeFiles
            ON  OrderPayments.CustomerKey = EmployeeFiles.EmployeeKey
WHERE  (OrderPayments.IsAccountSale = 1)
       AND (OrderPayments.PaymentDateTime >= @date1)
       AND (OrderPayments.PaymentDateTime <= @date2)
       AND (OrderPayments.LineDeleted = 0)
and OrderPayments.@BranchID and isnull(EmployeeFiles.FirstName,'')<>''
