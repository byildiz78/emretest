SELECT
	wb.BranchID AS [Şube Kodu],
	eb.BranchName AS [Şube Adı],
	wb.LogDateTime AS [Tarih],
	wb.StatusText AS [Şube Durumu],
	wb.Description AS [Açıklama] 
FROM
	webBranchStatusLogs AS wb
	INNER JOIN efr_Branchs AS eb ON wb.BranchID= eb.BranchID 
WHERE
	eb.IsActive= 1 
	AND wb.@BranchID
	AND wb.LogDateTime>@date1
	AND wb.LogDateTime<=@date2
