SELECT     SUM(OrderHeaders.AmountDue) AS Tutar, COUNT(OrderHeaders.OrderID) AS Adet, EmployeeFiles.FirstName AS DriverEmployeeName , 
                      EmployeeFiles.LastName
FROM         OrderHeaders LEFT OUTER JOIN
                      EmployeeFiles ON OrderHeaders.DriverEmployeeID = EmployeeFiles.EmployeeID AND OrderHeaders.BranchID = EmployeeFiles.BranchID

WHERE  LineDeleted = 0
       AND OrderDateTime >= @date1
       AND OrderDateTime <= @date2
      
       AND ORDERTYPE = 5
       AND employeefiles.EmployeeIsDriver=1
       AND OrderHeaders.@BranchID
and EmployeeFiles.@BranchID
GROUP BY EmployeeFiles.FirstName, EmployeeFiles.LastName
