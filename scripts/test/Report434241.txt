
SELECT OrderHeaders.OrderID               AS [Çek No],
       OrderHeaders.OrderDateTime         AS Tarih,
       OrderHeaders.StationID             AS [Terminal No],
             OrderHeaders.AmountDue             AS Tutar,
      CASE OrderHeaders.OrderStatus    
       
      WHEN  1 THEN 'AÇIK'
      WHEN 2 THEN  'KAPALI'
      else 'İPTAL'
            END  AS 'Durum',

       OrderHeaders.SpecificCustomerName  AS Müşteri,
       OrderHeaders.OrderNotes            AS [Not],
       OrderHeaders.DeleteReason          AS Açıklama,

       efr_Branchs.BranchName             AS [Şube Adı],
       OrderHeaders.EmployeeName AS Personel, 
       OrderHeaders.DiscountTotalAmount   AS [Toplam İndirim] ,
             OrderHeaders.LineDeleted           AS İptal
FROM   OrderHeaders
       LEFT OUTER JOIN efr_Branchs
            ON  OrderHeaders.BranchID = efr_Branchs.BranchID
            
       
            
WHERE OrderDateTime>=@date1
AND OrderHeaders.OrderDateTime<=@date2
AND orderheaders.@BranchID

ORDER BY OrderHeaders.OrderID asc
