
Select Top 20
  OrderTransactions.MenuItemText As Ürün,
  Sum(OrderTransactions.Quantity) As Miktar,( Sum(OrderTransactions.ExtendedPrice))  as Tutar
From
  OrderTransactions 
where 
  OrderTransactions.LineDeleted=0 and
OrderTransactions.OrderDateTime <=@date2 And
  OrderTransactions.OrderDateTime >=@date1
And OrderTransactions.@BranchID AND OrderTransactions.MenuItemUnitPrice<>0
Group By
  OrderTransactions.MenuItemText
Order By
  Sum(OrderTransactions.Quantity) asc
