SELECT
	br.BranchName AS [Şube],
	t.OrderDateTime AS [Tarih],
	t.OrderID AS [Çek No],
	t.MenuItemText AS [Ürün],
	t.IntegrationProductName AS [Eşleşmeyen Ürün]
FROM
	OrderTransactions t WITH ( NOLOCK ) 
	INNER JOIN efr_Branchs br WITH ( NOLOCK )  ON br.BranchID = t.BranchID
WHERE
	t.MenuItemText IN ( 'GETİR', 'TRENDYOL', 'WEB SİPARİŞ', 'YEMEK SEPETİ' )
	AND t.@BranchID
	AND t.OrderDateTime BETWEEN @date1 AND @date2
	AND t.LineDeleted = 0
	ORDER BY IntegrationProductName
