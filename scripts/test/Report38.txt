SELECT OrderTransactions.AddUserName AS Tipi, SUM(OrderTransactions.ExtendedPrice) AS Tutar 
FROM OrderTransactions 
INNER JOIN OrderHeaders AS h ON h.OrderKey = OrderTransactions.OrderKey AND h.BranchID = OrderTransactions.BranchID AND h.LineDeleted=0
WHERE 
OrderTransactions.OrderDateTime <=@date2 And
OrderTransactions.OrderDateTime >=@date1 
and OrderTransactions.@BranchID
AND OrderTransactions.LineDeleted=0 
GROUP BY OrderTransactions.AddUserName 
UNION ALL
SELECT OrderTransactions.AddUserName AS Tipi, count(distinct OrderTransactions.OrderKey) AS Tutar 
FROM OrderTransactions 
INNER JOIN OrderHeaders AS h ON h.OrderKey = OrderTransactions.OrderKey AND h.BranchID = OrderTransactions.BranchID AND h.LineDeleted=0
WHERE 
OrderTransactions.OrderDateTime <=@date2 And
OrderTransactions.OrderDateTime >=@date1 
and OrderTransactions.@BranchID 
AND OrderTransactions.LineDeleted=0 
GROUP BY OrderTransactions.AddUserName
