SELECT t.MenuItemText                AS Ürün,
       SUM(t.Quantity)               AS Adet,
       SUM(t.Quantity) * ISNULL(MenuItems.Barcode2, 1) AS KG,
       t.MenuItemGroupText           AS GRUP
FROM   OrderTransactions             AS t
       LEFT OUTER JOIN MenuItems
            ON  t.MenuItemKey = MenuItems.MenuItemKey
       LEFT OUTER JOIN OrderHeaders  AS h
            ON  t.OrderKey = h.OrderKey
WHERE  (t.LineDeleted = 0)
       AND t.OrderDateTime >= @date1
       AND t.OrderDateTime <= @date2
       AND t.@BranchID
       AND MenuItems.@BranchID
     and isnull(MenuItems.Barcode2,0)>1
GROUP BY
       t.MenuItemText,
       t.MenuItemGroupText,
       MenuItems.Barcode2
ORDER BY
       Ürün
