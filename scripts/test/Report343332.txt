SELECT 
	CONCAT(br.BranchName, ' (',br.BranchID,')') AS [ÅUBE],
	CONVERT( VARCHAR, h.OrderDateTime, 23) AS [TARÄ°H],
	COUNT ( * ) AS [TOPLAM]
FROM
	OrderHeaders h WITH (NOLOCK)
	INNER JOIN efr_Branchs br ON br.BranchID = h.BranchID
WHERE
h.LineDeleted=0
and	h.OrderDateTime>=@date1
and h.OrderDateTime<=@date2
and h.@BranchID
GROUP BY
	br.BranchID,
	br.BranchName,
	CONVERT ( VARCHAR, h.OrderDateTime, 23) 
ORDER BY
	CONVERT (	VARCHAR,h.OrderDateTime,	23)
