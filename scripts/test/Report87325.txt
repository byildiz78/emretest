DELETE FROM [efr_UserSessions] WHERE [IsExpired] = 1;
SELECT
	US.UserName AS [KULLANICI ADI],
	CONVERT(NVARCHAR(50),US.LoginDatetime,120) AS [GİRİŞ TARİHİ],
	US.Browser AS [TARAYICI],
	US.IpAddress AS [IP ADRESİ]
FROM
	[efr_UserSessions] US
	INNER JOIN efr_Users AS x ON x.UserID = US.UserID
WHERE
	US.[IsExpired] = 0
ORDER BY
	US.[LoginDatetime] DESC;
