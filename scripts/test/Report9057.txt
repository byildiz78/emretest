SELECT SUM(ot.Quantity) AS Adet,
       ot.MenuItemText AS [Ürün],
      datepart (hour, ot.TransactionDateTime) AS Saat,
       SUM(ot.ExtendedPrice) AS Tutar
FROM   OrderTransactions ot

WHERE ot.LineDeleted=0 AND ot.OrderDateTime>=@date1 AND ot.OrderDateTime<=@date2 and ot.@BranchID
GROUP BY
      datepart (hour, ot.TransactionDateTime),ot.MenuItemText
      
ORDER BY  datepart (hour, ot.TransactionDateTime) asc
