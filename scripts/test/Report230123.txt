
SELECT eb.BranchName AS Şube,
h.DocumentDateTime AS [Belge Tarihi],h.OrderSerialNo AS [Belge No],
pp.ProductCode AS [Stok kodu],
d.ProductName AS Ürün,
d.Incoming AS Miktar,
ppu.UnitName AS Birim,
pp.PurchasePrice1 AS [Birim Fiyat],
d.Incoming*pp.PurchasePrice1 AS Tutar


FROM   posOrderHeader h
    inner join posOrderDetail d
            ON  d.OrderKey = h.OrderKey
  INNER join efr_Branchs eb
            ON  eb.BranchID = h.BranchID
            
            INNER JOIN posProducts pp
            ON pp.ProductID=d.ProductID
            
            INNER JOIN posProductUnits ppu
            
            ON ppu.UnitID = d.UnitID
WHERE  h.OrderTypeID = 42
       AND h.RefDocumentNo = 'DIŞALIM'
       AND h.@BranchID
       AND h.DocumentDateTime>=@date1
       AND h.DocumentDateTime<=@date2
