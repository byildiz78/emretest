select
br.BranchName as Şube,
case when wb.Status=0 then 'Aktif'
when  wb.Status=2 then 'Yoğun' else 'Inaktif'
end as Durumu,
cast(DateDiff(dd, LastAccessDatetime, @date2) as varchar) + ' gün '+cast( DateDiff(hh, LastAccessDatetime, @date2)% 24 as varchar) + ' saat '+cast( DateDiff(mi, LastAccessDatetime, @date2) % 60 as varchar)  +' dakika' as Süre,
case when wb.Status=1 then wb.Description else '' end as [Kapanma Sebebi]

from efr_Branchs br
inner join pfTavukDunyasiWebOrder.dbo.webBranchStatus wb on br.BranchID=wb.BranchID
where br.IsActive=1
and br.@BranchID 
and  wb.LastAccessDatetime between @date1 and @date2

order by br.BranchName
