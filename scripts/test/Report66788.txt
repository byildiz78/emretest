SELECT
	b.BranchName AS [Şube],
	DATEADD( DAY, DATEDIFF( dd, 0, DATEADD( HOUR,- 0, h.OrderDateTime )), 0 ) AS Tarih,
	CONVERT ( nvarchar, DATEADD( HOUR,- 0, h.OrderDateTime ), 108 ) AS Saat,
	h.AddUserName AS Personel,
	isnull( h.DineInTableName, '' ) AS [Masa],
	h.GuestNumber AS [Kişi Sayısı],
	h.OrderID AS [Adisyon No],
	t.MenumItemCategoryText AS [Kategori],
	t.MenuItemGroupText AS [Grubu],
	t.MenuItemText AS [Ürün],
	round( t.ExtendedPrice/ t.Quantity, 2 ) AS [Net Fiyat],
	t.Quantity AS [Miktar],
	round( t.ExtendedPrice, 2 ) AS [Tutar] 
FROM
	OrderHeaders AS h with (nolock)
	INNER JOIN OrderTransactions AS t with (nolock) ON t.OrderKey = h.OrderKey AND t.LineDeleted= 0	AND t.Quantity> 0	AND t.MenuItemUnitPrice> 0
	INNER JOIN efr_Branchs AS b ON b.BranchID = h.BranchID 
WHERE
	h.LineDeleted= 0 
	AND h.OrderType= 1 
	AND h.SubTotal> 0 
	AND h.@BranchID 
	AND h.OrderDateTime>@date1 
	AND h.OrderDateTime<@date2
