SELECT 'Nakit' AS Tipi,
       Round (SUM(OrderPayments.AmountPaid),2) AS Tutar
FROM   OrderPayments
WHERE  OrderPayments.PaymentMethod = '1'
       AND  OrderPayments.PaymentDateTime >= @ilkTarih
                                            AND OrderPayments.PaymentDateTime <= @sonTarih
                                            AND OrderPayments.@BranchID
      
UNION ALL
