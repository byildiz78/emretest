SELECT OrderTransactions.MenuItemText AS Urun,OrderTransactions.MenuItemGroupText AS 
       Grubu,SUM(OrderTransactions.Quantity) [Satilan Adet],ISNULL(GlobalMenuItems.MenuItemCost, 0) AS 
       [Birim Maliyet],ISNULL( SUM(OrderTransactions.Quantity * GlobalMenuItems.MenuItemCost),0) AS [Maliyet Tutar],SUM(OrderTransactions.ExtendedPrice) AS 
       [Satis Tutar]
FROM OrderTransactions
LEFT OUTER JOIN GlobalMenuItems ON OrderTransactions.MenuItemKey =GlobalMenuItems.MenuItemKey
AND OrderTransactions.BranchID=GlobalMenuItems.BranchID
WHERE  OrderTransactions.TransactionDateTime >= @date1
  AND OrderTransactions.TransactionDateTime <= @date2
  AND OrderTransactions.LineDeleted = 0
  AND OrderTransactions.@BranchID
GROUP BY OrderTransactions.MenuItemText, GlobalMenuItems.MenuItemCost, 
      OrderTransactions.MenuItemGroupText
ORDER BY OrderTransactions.MenuItemGroupText ASC
