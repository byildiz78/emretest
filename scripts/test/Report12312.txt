SELECT 
efr_Branchs.Branchname as Şube,
OrderTransactions.MenuItemText[ÜRÜN],
p.ProductCode AS [Stok Kodu],
 OrderTransactions.OrderID [ÇEK NUMARASI],
 OrderTransactions.MenumItemCategoryText [GRUP],
 OrderTransactions.MenuItemGroupText [KATEGORİ],
 SUM(OrderTransactions.Quantity) [MİKTAR],
 OrderTransactions.MenuItemUnitPrice AS [Birim Fiyat],
 SUM(OrderTransactions.ExtendedPrice) AS Toplam
FROM OrderTransactions
INNER JOIN NewGlobalMenuItems AS m ON OrderTransactions.MenuItemKey = m.MenuItemKey
LEFT OUTER JOIN posProducts AS p ON p.ProductKey = m.MenuItemKey
LEFT OUTER JOIN efr_Branchs ON OrderTransactions.BranchID = efr_Branchs.BranchID
WHERE OrderTransactions.LineDeleted = 0 
 AND OrderTransactions.OrderDateTime>= @date1
 AND OrderTransactions.OrderDateTime<= @date2
 AND OrderTransactions.@BranchID
and OrderTransactions.MenuItemtext not like '%mesaj%'
GROUP BY
OrderTransactions.OrderID,
 OrderTransactions.MenuItemText,
 OrderTransactions.MenumItemCategoryText,
 OrderTransactions.MenuItemGroupText,
 OrderTransactions.MenuItemUnitPrice,
efr_Branchs.Branchname,p.ProductCode
ORDER BY
 OrderTransactions.OrderID,
 OrderTransactions.MenumItemCategoryText,
 OrderTransactions.MenuItemGroupText,
 OrderTransactions.MenuItemText,
 OrderTransactions.MenuItemUnitPrice DESC
