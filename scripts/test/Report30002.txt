SELECT ISNULL(efr_Branchs.BranchName,'TANIMSIZ' )AS [ŞUBE],
OrderTransactions.MenuItemText[ÜRÜN],
OrderTransactions.MenuItemGroupText [GRUP],
OrderTransactions.MenumItemCategoryText [KATEGORİ],
OrderTransactions.MenuItemUnitPrice [BİRİM FİYAT],
SUM(OrderTransactions.Quantity) [MİKTAR] 
FROM summaryOrderTransactions as OrderTransactions
LEFT OUTER JOIN efr_Branchs
ON OrderTransactions.BranchID = efr_Branchs.BranchID
WHERE OrderTransactions.LineDeleted=0 AND OrderTransactions.TransactionDateTime>=@ilkTarih AND OrderTransactions.TransactionDateTime<=@sonTarih and OrderTransactions.@BranchID
GROUP BY efr_Branchs.BranchName ,
OrderTransactions.MenuItemText,
OrderTransactions.MenuItemGroupText,
OrderTransactions.MenumItemCategoryText,
OrderTransactions.MenuItemUnitPrice
