select [Şube],[Tarih],Combo,[Ürün],[Grup],SUM([Adet]) as Adet from (
SELECT 
b.BranchName as [Şube],
convert(varchar,t.OrderDateTime,104) as [Tarih],
(SELECT TOP 1 MenuItemText FROM OrderTransactions t1 where t1.TransactionKey=t.MainMenuItemTransactionKey) as Combo,
t.MenuItemText as [Ürün],
ISNULL(p.Customfield12,p.CustomField13) as [Grup],
SUM(t.Quantity) as [Adet]
FROM OrderTransactions t
INNER JOIN posProducts p on p.ProductKey=t.MenuItemKey
INNER JOIN efr_Branchs b on b.BranchID=t.BranchID
WHERE 1=1
AND t.OrderDateTime between @date1 and @date2
AND t.@BranchID
AND t.linedeleted=0
AND MainMenuItemTransactionKey IS NOT NULL
GROUP BY b.BranchName,convert(varchar,t.OrderDateTime,104),t.MenuItemText,ISNULL(p.customfield12,p.CustomField13),t.MainMenuItemTransactionKey) AS TABLO
GROUP BY [Şube],[Tarih],Combo,[Ürün],[Grup]
order by [Şube],[Tarih],Combo,[Ürün],[Grup]
