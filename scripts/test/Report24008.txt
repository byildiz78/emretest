Select
  gsOrderHeaders.OrderDateTime As Tarih,
  gsOrderHeaders.AmountDue As Tutar,
  CustomerFiles.CustomerName As Müşteri,
  CustomerFiles.DeliveryAddress As Adres,
  CustomerFiles.DeliveryRemarks As Açıklama,
  CustomerFiles.PhoneNumber As Telefon,
  EmployeeFiles.FirstName As Personel
From
  (gsOrderHeaders Left Join
    CustomerFiles On CustomerFiles.CustomerID = gsOrderHeaders.CustomerID and CustomerFiles.BranchID = gsOrderHeaders.BranchID) Left Join
  EmployeeFiles On EmployeeFiles.EmployeeID = gsOrderHeaders.DriverEmployeeID and EmployeeFiles.BranchID = gsOrderHeaders.BranchID 
Where
  gsOrderHeaders.OrderType = '4'
and  gsOrderHeaders.OrderDateTime <=@sonTarih And
  gsOrderHeaders.OrderDateTime >=@ilkTarih and gsOrderHeaders.@BranchID
