SELECT     posProducts.ProductCode AS [Stok Kodu], posProducts.ProductName AS [Stok Adı], posPickList.PickValue AS Kategori, posPickList_1.PickValue AS Grup, 
                      posProductUnits.UnitName AS [Temel Birim], posProductUnits_1.UnitName AS [Satın Alma Birimi], posWarehouse.WarehouseName AS [Varsayılan Depo], 
                      posProductPrices.IncomingPrice AS [Karttaki Alış F.], posProductPrices.OutgoingPrice AS [Karttakş Satış F], ROUND(posProductPrices.LastIncomingPrice, 3) AS [Fatura Son Alış F], 
                      ROUND(posProductPrices.LastOutgoingPrice, 3) AS [Fatura Son Satış F], 
                       posProducts.Notes AS Açıklama
FROM         posWarehouse RIGHT OUTER JOIN
                      posProductUnits AS posProductUnits_1 RIGHT OUTER JOIN
                      posProducts ON posProductUnits_1.UnitID = posProducts.UnitID2 LEFT OUTER JOIN
                      posProductUnits ON posProducts.UnitID = posProductUnits.UnitID ON posWarehouse.WarehouseID = posProducts.DefaultWarehouseID LEFT OUTER JOIN
                      posProductPrices ON posProducts.ProductID = posProductPrices.ProductID LEFT OUTER JOIN
                      posPickList ON posProducts.CategoryID = posPickList.ListID LEFT OUTER JOIN
                      posPickList AS posPickList_1 ON posProducts.GroupID = posPickList_1.ListID
WHERE     (ISNULL(posProducts.IsSaleProduct, 0) = 0)
