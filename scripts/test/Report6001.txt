SELECT     posPickList.PickValue AS Kategori, posCurrents.CurrentID, posCurrents.CurrentCode, posCurrents.CurrentName, posCurrents.CurrentFullName, 
                      posCurrents.CurrentKey, posCurrents.TaxOfficeName, posCurrents.TaxOfficeCode, posCurrents.TaxNumber, posCurrents.TotalDebt, posCurrents.TotalCredit, 
                      posCurrents.CurrencyID, posCurrents.CurrentTypeID, posCurrents.CategoryID, posCurrents.IsActive, posCurrents.Notes, posCurrents.CreditDayLimit, 
                      posCurrents.AllowMail, posCurrents.AllowSms, posCurrents.ExternalCode, posCurrents.BranchCode, posCurrents.IsBranch, posCurrents.CustomField1, 
                      posCurrents.CustomField2, posCurrents.CustomField3, posCurrents.CustomField4, posCurrents.CustomField5, posCurrents.CustomField6, posCurrents.CustomField7, 
                      posCurrents.CustomField8, posCurrents.CustomField9, posCurrents.CustomField10, posCurrents.AddUserID, posCurrents.AddDateTime, posCurrents.EditUserID, 
                      posCurrents.EditDateTime, posPickList_1.PickValue AS [Cari Tipi]
FROM         posCurrents LEFT OUTER JOIN
                      posPickList ON posCurrents.CategoryID = posPickList.ListID LEFT OUTER JOIN
                      posPickList AS posPickList_1 ON posCurrents.CurrentTypeID = posPickList_1.ListID
                      
                      
                      where (posCurrents.TotalDebt- posCurrents.TotalCredit) >0
                      
                      order by (posCurrents.TotalDebt- posCurrents.TotalCredit) desc
