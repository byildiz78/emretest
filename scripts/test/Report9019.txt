SELECT ot.MenuItemText,
       SUM(ot.Quantity) AS MIKTAR,
       MenuModifiers.MenuModifierText
FROM   OrderTransactions AS ot
       INNER JOIN MenuModifiers
            ON  ot.Mod1ID = MenuModifiers.MenuModifierID
WHERE  ot.LineDeleted = 0
       AND ot.Mod1ID > 0
       AND OT.TransactionDateTime >= @ilkTarih
           AND TransactionDateTime <= @SonTarih  AND ot.@BranchID 
GROUP BY
       OT.MenuItemText,
       MenuModifiers.MenuModifierText

UNION ALL
