Select
  'Yüklenen Para Puan' As Tipi,
  Sum(CustomerCredits.CreditAmount) As Tutar
From
  CustomerCredits 
        Where CustomerCredits.CreditDateTime <=@sonTarih  And  CustomerCredits.CreditDateTime >=@ilkTarih and  CustomerCredits.@BranchID

UNION ALL
SELECT 'Tüketilen Para Puan' as Tipi,
  Sum(gsOrderHeaders.CreditAmountUsed) As Tutar
From
  (gsOrderHeaders Inner Join
    CustomerCredits On CustomerCredits.CreditID = gsOrderHeaders.CreditID)
  Inner Join
  CustomerFiles On CustomerFiles.CustomerID = CustomerCredits.CustomerID
  Where gsOrderHeaders.OrderDateTime <=@sonTarih And  gsOrderHeaders.OrderDateTime >=@ilkTarih and  gsorderheaders.OrderStatus <>'3'
  and gsOrderHeaders.@BranchID
