select 
eb.BranchName as [Sube Adı],
cast(h.OrderID as varchar) as [Çek No],
h.OrderTypeSourceExternalNo as [Web Sipariş No],
h.CustomerName as [Müşteri Adı],
h.AmountDue as [Çek Tutarı],
CASE 
wHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=1 then 'GARSON'
wHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=2 then 'EL TERMİNALİ'
WHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=3 then 'WEB SİPARİŞ'
wHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=4 then 'ÇAĞRI MERKEZİ'
wHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=5 then 'QR MENÜ'
wHEN h.OrderTypeSourceExternalNo LIKE '%web%' and OrderSource=6 then 'YEMEK SEPETİ'
Else 'BİLİNMİYOR' end [Sipariş Kanalı],
Case h.OrderType
When '1' then 'MASA'
When '2' then 'İSME ÇEK'
When '3' then 'AL GÖTÜR'
When '4' then 'TEZGAH'
When '5' then 'PAKET SATIŞ'
Else 'BİLİNMİYOR' end AS [Satış  Tipi],
OrderDateTime as [Sipariş Tarihi],
CASE WHEN h.Linedeleted=1  THEN 'İptal' ELSE 'Aktif' END 'Durum'
from OrderHeaders as h 
INNER JOIN efr_Branchs as eb on h.BranchID=eb.BranchID
where (h.OrderTypeSourceExternalNo LIKE '%Web%')
AND h.OrderDateTime between @date1 and @date2
AND h.@BranchID
