DECLARE @startDate DATETIME;set @startDate=GETDATE();
IF DATEPART(hour,@startDate)<6
 set @startDate=DATEADD(dd, -1, DATEDIFF(dd, 0, @startDate))
ELSE
set @startDate=DATEADD(dd, 0, DATEDIFF(dd, 0, @startDate))
Select
  OrderHeaders.OrderDateTime As Tarih,
  OrderHeaders.AmountDue As Tutar,
  CustomerFiles.CustomerName As Müşteri,
  CustomerFiles.DeliveryAddress As Adres,
  CustomerFiles.DeliveryRemarks As Açıklama,
  CustomerFiles.PhoneNumber As Telefon,
  EmployeeFiles.FirstName As Personel
From
  (OrderHeaders Left Join
    CustomerFiles On CustomerFiles.CustomerID = OrderHeaders.CustomerID and CustomerFiles.BranchID = OrderHeaders.BranchID) Left Join
  EmployeeFiles On EmployeeFiles.EmployeeID = OrderHeaders.DriverEmployeeID and EmployeeFiles.BranchID = OrderHeaders.BranchID 
WHERE  OrderHeaders.OrderType = '4' and  OrderHeaders.OrderDateTime >=@startDate and OrderHeaders.@BranchID
