SELECT b.BranchName AS Şube,
CASE WHEN h.OrderType=1 THEN 'Masa'
WHEN h.OrderType=2 THEN 'Masa'
WHEN h.OrderType=3 THEN 'İç Paket'
WHEN h.OrderType=4 THEN 'Tezgah Satış'
WHEN h.OrderType=5 THEN 'Paket'
ELSE '' END  AS Tipi,SUM(h.AmountDue) AS Tutar,COUNT(h.AutoID) AS Adet
FROM OrderHeaders AS h WITH (NOLOCK)
INNER JOIN efr_Branchs AS b ON b.BranchID=h.BranchID
WHERE h.OrderDateTime>@date1
AND h.OrderDateTime<@date2
AND h.LineDeleted=0
AND h.@BranchID
GROUP BY h.BranchID,h.OrderType,b.BranchName
ORDER BY b.BranchName
