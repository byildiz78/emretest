
SELECT    CONVERT(VARCHAR,PaymentDateTime, 103) AS Tarih,PaymentMethodName AS [Ödeme Tipi],
       SUM(AmountPaid) AS Tutar
FROM   OrderPayments

WHERE LineDeleted=0 AND OrderPayments.@BranchID AND PaymentDateTime>=@date1 AND PaymentDateTime<=@date2

GROUP BY PaymentMethodName,CONVERT(VARCHAR,PaymentDateTime, 103)

ORDER BY CONVERT(VARCHAR,PaymentDateTime, 103) desc
