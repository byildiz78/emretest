SELECT OrderHeaders.OrderDateTime AS TARİH,
       OrderHeaders.OrderID AS [ÇEK NO],
       OrderTransactions.MenuItemText as ÜRÜN,
      MenuItems.DefaultUnitPrice AS [TANIMLI FİYAT TUTAR],
       OrderTransactions.MenuItemUnitPrice AS [SATILAN FİYAT TUTAR]
FROM   MenuItems
       RIGHT OUTER JOIN OrderTransactions
            ON  MenuItems.MenuItemKey = OrderTransactions.MenuItemKey
       LEFT OUTER JOIN OrderHeaders
            ON  OrderTransactions.OrderKey = OrderHeaders.OrderKey
WHERE  MenuItems.DefaultUnitPrice <> OrderTransactions.MenuItemUnitPrice
       AND OrderHeaders.LineDeleted = 0
       AND OrderTransactions.LineDeleted = 0
       AND OrderTransactions.MenuItemUnitPrice<>0
       AND OrderHeaders.OrderDateTime>=@date1
       AND OrderHeaders.OrderDateTime<=@date2
and OrderTransactions.@BranchID 
 AND OrderHeaders.@BranchID
AND MenuItems.@BranchID
