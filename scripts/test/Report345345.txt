
SELECT SUM(OrderHeaders.AmountDue)  AS [Ciro KDV Dahil],
round (SUM(OrderHeaders.AmountDue)/108*100,2
) AS [Net Ciro],
       efr_Branchs.BranchName       AS Sube,
       CONVERT(VARCHAR, OrderHeaders.OrderDateTime, 103) AS Gun,
       '' AS [Ciro Primi]
FROM   OrderHeaders
       INNER JOIN efr_Branchs
            ON  OrderHeaders.BranchID = efr_Branchs.BranchID
WHERE  OrderHeaders.LineDeleted = 0
 AND OrderHeaders.OrderDateTime >= @date1
   AND OrderHeaders.OrderDateTime <= @date2
and  OrderHeaders.@BranchID
       
GROUP BY  efr_Branchs.BranchName  ,CONVERT(VARCHAR, OrderHeaders.OrderDateTime, 103)
