SELECT orderId AS [Çek No],
OrderDateTime AS Tarih,
AmountDue AS Tutar,
DineInTableName AS [Güncel Masa No],
OrderNotes AS [Önceki Masa],
AddUserName AS [Ekleyen Kullanıcı],
EditUserName [Son Düzenleyen]
FROM OrderHeaders 
WHERE LineDeleted = 0 
AND OrderNotes LIKE '%(%'
AND OrderDateTime >= @date1
AND OrderDateTime <= @date2
AND OrderHeaders.@BranchID
