select
b.BranchName,
CONVERT(DATE,t.OrderDateTime) as Tarih,
CASE ISNULL(h.OrderType,6)
WHEN '1' THEN 'MASA SERVIS'
WHEN '2' THEN 'İSME ÇEK'
WHEN '3' THEN 'AL GÖTÜR'
WHEN '4' THEN 'TEZGAH'
WHEN '5' THEN 'PAKET SERVIS'
ELSE 'TANIMSIZ'
END AS 'Satış Türü',
t.MenuITemtext as [Ürün],
SUM(CASE WHEN t.customfield1 IS NOT NULL THEN t.Quantity ELSE 0 END) AS [Kombo Tabak],
SUM(CASE WHEN t.customfield1 IS NULL THEN t.Quantity ELSE 0 END) AS [Kombosuz Tabak],
SUM(t.Quantity) as Toplam
from OrderTransactions t
LEFT JOIN OrderHeaders h with(nolock) on h.OrderKey=t.OrderKey
LEFT JOIN efr_Branchs b with(nolock) on b.BranchID=t.BranchID
INNER JOIN posProducts as m with(nolock) on m.ProductKey=t.MenuItemKey
where t.OrderDateTime BETWEEN @date1 and @date2 and t.LineDeleted=0 and m.CustomField12='TABAKLAR' and b.@BranchID
GROUP BY b.BranchName,CONVERT(DATE,t.OrderDateTime),h.OrderType,t.MenuITemtext
order by b.BranchName
