Select Top 10
  MenuItems.MenuItemText As Tipi,
  Sum(OrderTransactions.Quantity) As Tutar
From
  (OrderTransactions Inner Join
    MenuItems On MenuItems.MenuItemID = OrderTransactions.MenuItemID and MenuItems.BranchID = OrderTransactions.BranchID) Inner Join
  OrderHeaders On OrderHeaders.OrderID = OrderTransactions.OrderID and OrderHeaders.BranchID = OrderTransactions.BranchID 
where 
  OrderTransactions.TransactionStatus = '1' and
OrderHeaders.OrderDateTime <=@sonTarih And
  OrderHeaders.OrderDateTime >=@ilkTarih
And OrderTransactions.@BranchID AND MenuItems.MenuCategoryID in(14,15)
Group By
  MenuItems.MenuItemText
Order By
  Sum(OrderTransactions.Quantity) Desc
